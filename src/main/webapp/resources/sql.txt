CREATE TABLE MEMBER(
	ID VARCHAR2(20) PRIMARY KEY,
	PW VARCHAR2(100) NOT NULL,
	NAME VARCHAR2(10) NOT NULL,
	ADDR VARCHAR2(100) NOT NULL,
	EMAIL VARCHAR2(30) NOT NULL UNIQUE,
	PHONE VARCHAR2(20) NOT NULL,
	REGDATE DATE DEFAULT SYSDATE
);
